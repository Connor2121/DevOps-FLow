$cert = New-Object System.Security.Cryptography.X509Certificates.X509Certificate2(
    "C:\path\to\combined.pfx", 
    $password
)

# If this prints details, the PFX is valid
Write-Output $cert.Subject
Write-Output $cert.Thumbprint
Write-Output $cert.NotAfter


$pfxBytes = [System.IO.File]::ReadAllBytes("C:\path\to\combined.pfx")
$pfxB64 = [Convert]::ToBase64String($pfxBytes)

# View it
Write-Output $pfxB64

# Or copy directly to clipboard
$pfxB64 | Set-Clipboard
