import os
import httpx
import uvicorn
from fastmcp import FastMCP
from databricks.sdk import WorkspaceClient
from databricks.sdk.service.sql import StatementState

mcp = FastMCP(name="test-mcp-server")

# One shared client — picks up the app's service principal auth automatically
_ws = WorkspaceClient()

DEFAULT_WAREHOUSE_ID = os.environ.get("DATABRICKS_WAREHOUSE_ID")


@mcp.tool
def word_count(text: str) -> dict:
    """Count words and characters in the given text.

    Args:
        text: The input string to analyze.
    """
    words = text.split()
    return {"word_count": len(words), "char_count": len(text)}


@mcp.tool
def run_sql(query: str, warehouse_id: str | None = None, max_rows: int = 100) -> dict:
    """Execute a SQL query on a Databricks SQL warehouse and return the results.

    Use this when you already know the SQL you want to run. For natural-language
    questions about data, prefer the Genie tool instead.

    Args:
        query: The SQL statement to execute.
        warehouse_id: SQL warehouse ID. Falls back to DATABRICKS_WAREHOUSE_ID env var.
        max_rows: Row cap (default 100, hard ceiling 1000).

    Returns:
        dict with 'columns', 'rows', and 'row_count', or 'error' on failure.
    """
    wh = warehouse_id or DEFAULT_WAREHOUSE_ID
    if not wh:
        return {"error": "No warehouse_id provided and DATABRICKS_WAREHOUSE_ID not set."}

    capped = min(max_rows, 1000)

    try:
        resp = _ws.statement_execution.execute_statement(
            warehouse_id=wh,
            statement=query,
            wait_timeout="30s",
            row_limit=capped,
        )
        if resp.status.state != StatementState.SUCCEEDED:
            msg = resp.status.error.message if resp.status.error else str(resp.status.state)
            return {"error": msg}

        columns = [c.name for c in (resp.manifest.schema.columns or [])]
        rows = resp.result.data_array or []
        return {"columns": columns, "rows": rows, "row_count": len(rows)}
    except Exception as e:
        return {"error": str(e)}


@mcp.tool
def get_table_schema(full_name: str) -> dict:
    """Get the schema of a Unity Catalog table.

    Call this before run_sql to discover column names, types, and descriptions.

    Args:
        full_name: Three-part UC name, e.g. 'main.sales.orders'.

    Returns:
        dict with the table's columns (name, type, comment) and table comment.
    """
    try:
        table = _ws.tables.get(full_name)
        columns = [
            {"name": c.name, "type": c.type_text, "comment": c.comment or ""}
            for c in (table.columns or [])
        ]
        return {
            "full_name": table.full_name,
            "comment": table.comment or "",
            "columns": columns,
        }
    except Exception as e:
        return {"error": str(e)}


@mcp.tool
def get_weather(latitude: float, longitude: float) -> dict:
    """Get current weather for a location using the free Open-Meteo API.

    Args:
        latitude: Decimal degrees (e.g. 35.9 for Manteo, NC).
        longitude: Decimal degrees (e.g. -75.67 for Manteo, NC).

    Returns:
        Current temperature (C), humidity (%), wind speed (km/h), weather code, and timestamp.
    """
    try:
        resp = httpx.get(
            "https://api.open-meteo.com/v1/forecast",
            params={
                "latitude": latitude,
                "longitude": longitude,
                "current": "temperature_2m,relative_humidity_2m,weather_code,wind_speed_10m",
            },
            timeout=10.0,
        )
        resp.raise_for_status()
        current = resp.json().get("current", {})
        return {
            "temperature_c": current.get("temperature_2m"),
            "humidity_pct": current.get("relative_humidity_2m"),
            "wind_speed_kmh": current.get("wind_speed_10m"),
            "weather_code": current.get("weather_code"),
            "time": current.get("time"),
        }
    except Exception as e:
        return {"error": str(e)}


def main():
    port = int(os.environ.get("DATABRICKS_APP_PORT", 8000))
    uvicorn.run(mcp.http_app(), host="0.0.0.0", port=port)


if __name__ == "__main__":
    main()
