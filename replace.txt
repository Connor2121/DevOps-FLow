def generate_sharepoint_online_auth(dbutils: IDBUtils) -> Dict[str, str]:
    credential = ClientSecretCredential(
        tenant_id=get_secrets(dbutils, "barings-tenant-id", "kv-datahub"),
        client_id=get_secrets(dbutils, "sharepointClientId", "kv-datahub"),
        client_secret=get_secrets(dbutils, "sharepointClientSecret", "kv-datahub"),
    )
    oauth_token = credential.get_token("https://test.sharepoint.com/.default").token

    headers = {
        "Authorization": "Bearer " + oauth_token,
        "Accept": "application/json;odata=verbose",
        "Content-Type": "application/json;odata=verbose",
    }

    return headers
